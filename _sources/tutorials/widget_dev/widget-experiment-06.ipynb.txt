    "def interact(tpf, lc=None):\n",
    "    \"\"\"\n",
    "    Interact with a linked target pixel file and lightcurve\n",
    "\n",
    "    Parameters\n",
    "    ----------\n",
    "    lc : an optional user-supplied pre-processed lightcurve for this target\n",
    "\n",
    "    Returns\n",

    "def my_custom_lc_cleaner(tpf, cleaning='best'):\n",
    "    '''Returns a clean lightcurve given an input tpf'''\n",
    "    \n",
    "    nan_mask = np.all(np.isnan(tpf.flux), axis=0)\n",
    "    empty_mask = tpf.pipeline_mask * False\n",
    "    aper_mask = empty_mask\n",
    "    aper_mask[1:-1, 1:-1] = True\n",
    "    back_mask = ~aper_mask & ~nan_mask\n",
    "    \n",
    "    lc_aper = tpf.to_lightcurve(aperture_mask=aper_mask)/aper_mask.sum()\n",
    "    lc_back = tpf.to_lightcurve(aperture_mask=back_mask)/back_mask.sum()\n",
    "    \n",
    "    lc_bg_sub = lc_aper - lc_back.flux\n",
    "    \n",
    "    # Flag outliers\n",
    "    _, outliers1 = lc_aper.remove_outliers(return_mask=True)\n",
    "    _, outliers2 = lc_back.remove_outliers(return_mask=True)\n",
    "\n",
    "    _, outliers3 = lc_aper.flatten().remove_outliers(return_mask=True)\n",
    "\n",
    "    outlier_dict= {'none': outliers1*False,\n",
    "                   'safe':(tpf.quality > 0.0),\n",
    "                   'soft': outliers1,\n",
    "                   'medium':(outliers1 | outliers2),\n",
    "                   'best': (outliers1 | outliers2 | (tpf.quality > 0.0)),\n",
    "                   'aggressive':(outliers1 | outliers2 | (tpf.quality > 0.0) | outliers3)}\n",
    "\n",
    "    outliers = outlier_dict[cleaning]\n",
    "    \n",
    "    lc_cln = lc_bg_sub[~outliers]\n",
    "    \n",
    "    return lc_cln"
